    Student ||--|| Selection : "1 选题 1"
    Topic ||--o{ Selection : "1 被选 N"
    Selection ||--o{ Submission : "1 提交 N"
    Submission ||--|| Review : "1 审阅 1"
    Teacher ||--o{ Review : "1 评审 N"

    %% ==========================================
    %% 实体属性定义
    %% ==========================================

    Department {
        int dept_id PK "系ID【主键】"
        varchar dept_name "系名称【唯一】"
        datetime created_at "创建时间"
    }

    Major {
        int major_id PK "专业ID【主键】"
        varchar major_name "专业名称"
        int dept_id FK "系ID【外键】"
        datetime created_at "创建时间"
    }

    Class {
        int class_id PK "班级ID【主键】"
        varchar class_name "班级名称"
        int major_id FK "专业ID【外键】"
        varchar grade "年级"
        datetime created_at "创建时间"
    }

    Student {
        varchar stu_id PK "学号【主键】"
        varchar stu_name "姓名"
        varchar password "密码【BCrypt加密】"
        int class_id FK "班级ID【外键】"
        varchar phone "联系电话"
        varchar email "邮箱"
        datetime created_at "创建时间"
    }

    Teacher {
        varchar teacher_id PK "工号【主键】"
        varchar teacher_name "姓名"
        varchar password "密码【BCrypt加密】"
        int dept_id FK "系ID【外键】"
        varchar title "职称"
        varchar research_direction "研究方向"
        varchar phone "联系电话"
        varchar email "邮箱"
        int max_students "最大指导学生数"
        datetime created_at "创建时间"
    }

    Topic {
        int topic_id PK "题目ID【主键】"
        varchar title "题目名称"
        text description "题目描述"
        varchar teacher_id FK "教师工号【外键】"
        varchar direction "研究方向"
        tinyint status "状态【0可选/1已选】"
        int max_select "最多可选人数"
        datetime created_at "创建时间"
    }

    Selection {
        int selection_id PK "选题ID【主键】"
        varchar stu_id FK_u "学号【外键，唯一】"
        int topic_id FK "题目ID【外键】"
        datetime select_time "选题时间"
        varchar status "状态【进行中/已完成】"
        decimal final_score "最终综合成绩"
        datetime completed_at "完成时间"
    }

    Submission {
        int submission_id PK "提交ID【主键】"
        int selection_id FK "选题记录ID【外键】"
        varchar stage "阶段【开题/中期/初稿/终稿】"
        varchar file_path "文档路径"
        text description "提交说明"
        datetime submit_time "提交时间"
        varchar status "审阅状态"
        int version "版本号"
    }

    Review {
        int review_id PK "审阅ID【主键】"
        int submission_id FK_u "提交ID【外键，唯一】"
        varchar teacher_id FK "审阅教师工号【外键】"
        decimal score "评分【0-100】"
        text comment "评语"
        datetime review_time "审阅时间"
    }

    Deadline {
        int deadline_id PK "期限ID【主键】"
        varchar stage "阶段名称"
        date start_date "开始日期"
        date end_date "截止日期"
        varchar description "说明"
        varchar semester "学期"
    }

    AuditLog {
        bigint log_id PK "日志ID【主键】"
        varchar table_name "操作表名"
        varchar operation "操作类型【增删改】"
        varchar record_id "被操作记录ID"
        json old_value "旧值【JSON】"
        json new_value "新值【JSON】"
        varchar operated_by "操作者"
        datetime operation_time "操作时间"
        varchar ip_address "IP地址"
    }

    SysUser {
        int user_id PK "用户ID【主键】"
        varchar username "用户名【唯一】"
        varchar password "密码【BCrypt加密】"
        varchar real_name "真实姓名"
        varchar role "角色"
        datetime created_at "创建时间"
    }

📋 图例说明

PK = Primary Key（主键）— 唯一标识每条记录
FK = Foreign Key（外键）— 引用另一张表的主键
普通字段 — 业务数据属性
🔑 关键设计说明：
  • Student — Selection (1:1)：每位学生只能选择 一个 毕业设计题目（selection.stu_id 设 UNIQUE 约束）
  • Topic — Selection (1:N)：一个题目可被 多位 学生选择（topic.max_select 控制上限）
  • Submission — Review (1:1)：每份提交只能被 一次 审阅（review.submission_id 设 UNIQUE 约束）
  • 全文索引：topic.title + topic.description 使用 MySQL ngram 解析器支持中文全文搜索
  • 分区表：audit_log 按 operation_time 的年份进行 RANGE 分区
  • 密码安全：学生/教师/管理员密码均使用 BCrypt 单向加密存储
  • 辅助表：Deadline、AuditLog、SysUser 为独立辅助表，不参与核心业务关联
